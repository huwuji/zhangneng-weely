    function test(m,n){
        if(m===0||n===0){
            return 0;
        }
        if(m===1||n===1){
            return 1;
        }
        if(m!==1||n!==1){

        return 2*test(m-1,n-1)+test(m-2,n)+test(m,n-2);
        }
    }

    return test(m,n);
};
