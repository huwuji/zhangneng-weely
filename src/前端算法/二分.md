一个长度为n-1的递增排序数组中的所有数字都是唯一的，并且每个数字都在范围0～n-1之内。在范围0～n-1内的n个数字中有且只有一个数字不在该数组中，请找出这个数字

function miss(nums){
	let start=0;
	let end=nums.length;
	if(start!==nums[0]){
		return start;
	}
	if(end!==nums[nums.length-1]){
		return end;
	}
	while(start<end){
	let mid =Math.floor((start+end)/2);
	if(mid<nums[mid]){
		//mid前面缺少数字
		end--;
	}
	if(mid===nums[mid]){
	start++;
	}
	}
	return start;
	
}

function newMiss(nums){
	let start=0;
	let end=nums.length;
	if(start!==nums[0]){
		return start;
	}
	if(end!==nums[nums.length-1]){
		return end;
	}
	while(start<end){
	let mid =Math.floor((start+end)/2);
	if(mid===start){
	return start+1;
	}
	if(mid<nums[mid]){
		//mid前面缺少数字
		end=mid;
	}
	if(mid===nums[mid]){
	start=mid;
	}
	}
	return start;
}
